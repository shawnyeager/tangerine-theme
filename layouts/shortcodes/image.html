{{/* Purpose: Renders responsive images with AVIF/WebP support and lazy loading Parameters: - src (string, required): Image path (page bundle or site resources) - alt (string, required): Alt text for accessibility - variant (string, optional): CSS class for styling variants - lazy (string, optional): Enable lazy loading (default: "true") Context Expected: - .Page.Resources - Page bundle resources - resources.Get - Site-level resources Features: - Automatic WebP conversion with fallback to original format - Responsive srcset with 3 sizes: 640px, 1024px, 1920px - HTML element for modern format support - Format priority: WebP (smaller) → Original (fallback) - Lazy loading by default - Width and height attributes for layout stability - Falls back to basic if resource not found Usage: {{< image src="photo.jpg" alt="Description of photo" >}} {{< image src="photo.jpg" alt="Description" variant="full-width" >}} {{< image src="photo.jpg" alt="Description" lazy="false" >}} Important Notes: - Image must exist in page bundle or site resources - Hugo processes images automatically (creates AVIF, WebP, resizes) - Sizes attribute optimized for 700px max-width container - If image not found as resource, falls back to basic img with src path - Browser support: WebP (96%+), original (100%) */}} {{- $src := .Get "src" -}} {{- $alt := .Get "alt" -}} {{- $variant := .Get "variant" | default "" -}} {{- $lazy := .Get "lazy" | default "true" -}} {{- /* Validate required parameters */ -}} {{- if not $src -}} {{- errorf "image shortcode: 'src' parameter is required in %s" .Position -}} {{- end -}} {{- if not $alt -}} {{- errorf "image shortcode: 'alt' parameter is required for accessibility in %s" .Position -}} {{- end -}} {{- /* Try to get image resource for processing */ -}} {{- $image := .Page.Resources.GetMatch $src -}} {{- if not $image -}} {{- /* Try site resources */ -}} {{- $image = resources.Get $src -}} {{- end -}} {{- if $image -}} {{- /* Determine image sizes based on variant */ -}} {{- $isCompact := or (eq $variant "headshot") (eq $variant "portrait") -}} {{- $small := $image.Resize (cond $isCompact "280x" "640x") -}} {{- $medium := $image.Resize (cond $isCompact "560x" "1024x") -}} {{- $smallAVIF := $image.Resize (cond $isCompact "280x avif" "640x avif") -}} {{- $mediumAVIF := $image.Resize (cond $isCompact "560x avif" "1024x avif") -}} {{- $smallWebP := $image.Resize (cond $isCompact "280x webp" "640x webp") -}} {{- $mediumWebP := $image.Resize (cond $isCompact "560x webp" "1024x webp") -}} {{- $sizes := cond $isCompact "280px" "(max-width: 640px) 100vw, (max-width: 1024px) 90vw, 1024px" -}} {{- /* AVIF - smallest file size, modern browser support */ -}} {{- /* WebP - wide browser support fallback */ -}} {{- /* Original format - universal fallback */ -}} {{- else -}} {{- /* Fallback to basic img tag with lazy loading */ -}} {{- end -}}
